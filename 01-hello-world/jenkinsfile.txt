pipeline {
   agent any
  stages {
     stage('Run gcloud version') {
        steps {
           sh 'gcloud --version'
        }
     }
      
      stage('set project') {
        steps {
           sh 'gcloud config set project silent-thunder-256105'
        }
     }
     stage('list project') {
        steps {
           sh 'gcloud config list project'
        }
     }
     stage('zoneset') {
        steps {
           sh 'gcloud config set compute/zone us-central1-b'
        }
     }
     stage('compute VM create') {
        steps {
           sh 'gcloud compute instances create test'
        }
     }
  }
}
